\section{Analytic Geometry}

\subsection{Lines}

\begin{align}
	\Delta x &= x_2 - x_1	\\
	\Delta y &= y_2 - y_1	\\
	m &= \frac{ \Delta x }{ \Delta y }
\end{align}

\subsection{Distance Between Two Points; Circles}

\begin{align}
	\text{distance in } \mathbb{R}^2 &= \sqrt{(\Delta x)^2 + (\Delta y)^2}	\\
	\text{distance in } \mathbb{R}^3 &= \sqrt{(\Delta x)^2 + (\Delta y)^2 + (\Delta z)^2}
\end{align}
